<script>

// 計時器

const start = new Date(2025, 4, 28);

function updateTimer() {

const diff = new Date() - start;

const d = Math.floor(diff / 86400000);

const h = Math.floor((diff / 3600000) % 24);

const m = Math.floor((diff / 60000) % 60);

const s = Math.floor((diff / 1000) % 60);

document.getElementById('count').innerHTML = `${d}d ${h}h ${m}m ${s}s`;

}

setInterval(updateTimer, 1000); updateTimer();

 

// Wiki 切換

function openWiki(evt, name) {

let content = document.querySelectorAll(".wiki-section .content-pane");

content.forEach(p => { p.style.display = "none"; p.classList.remove("active"); });

let tabs = document.querySelectorAll(".wiki-section .tab-btn");

tabs.forEach(t => t.classList.remove("active"));

document.getElementById(name).style.display = "block";

document.getElementById(name).classList.add("active");

evt.currentTarget.classList.add("active");

document.getElementById('wikiScroll').scrollTop = 0;

}

 

// 人物切換

function showChar(evt, name) {

let content = document.querySelectorAll(".char-container .char-content");

content.forEach(p => { p.style.display = "none"; p.classList.remove("active"); });

let navs = document.querySelectorAll(".char-node");

navs.forEach(n => n.classList.remove("active"));

let target = document.getElementById(name);

target.style.display = (name === 'support') ? "block" : "flex";

target.classList.add("active");

evt.currentTarget.classList.add("active");

document.getElementById('charScroll').scrollTop = 0;

}

 

// 音樂播放

const audio = document.getElementById('audio');

const playBtn = document.getElementById('playBtn');

function togglePlay() {

if (audio.paused) { audio.play(); playBtn.innerText = '❚❚'; }

else { audio.pause(); playBtn.innerText = '▶'; }

}
